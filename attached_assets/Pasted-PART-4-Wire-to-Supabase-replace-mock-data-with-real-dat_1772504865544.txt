function timeAgoFromDate(dateStr){
  if(!dateStr) return '—';
  const now = new Date();
  const d = new Date(dateStr);
  const secs = Math.floor((now - d) / 1000);
  if(secs < 60) return 'just now';
  const mins = Math.floor(secs / 60);
  if(mins < 60) return `${mins}m`;
  const hrs = Math.floor(mins / 60);
  if(hrs < 24) return `${hrs}h`;
  const days = Math.floor(hrs / 24);
  if(days < 7) return `${days}d`;
  if(days < 30) return `${Math.floor(days/7)}w`;
  if(days < 365) return `${Math.floor(days/30)}mo`;
  return `${Math.floor(days/365)}y`;
}

jsasync function submitPost(){
  const textEl = document.getElementById('compose-text');
  const text = textEl?.value?.trim();
  if(!text){ closeModal('modal-compose'); return; }

  // Get selected category
  const selectedPill = document.querySelector('#compose-cat-pills .cat-pill.selected');
  const category = selectedPill ? Object.keys(CAT).find(k => CAT[k].label === selectedPill.textContent.trim().slice(2)) || 'nature' : 'nature';

  const btn = document.querySelector('#modal-compose .btn-primary');
  if(btn){ btn.disabled = true; btn.textContent = 'Posting…'; }

  try {
    if(currentUser){
      const { error } = await sb.from('posts').insert({
        user_id: currentUser.id,
        content: text,
        category: category,
      });
      if(error) throw error;
    }

    closeModal('modal-compose');
    textEl.value = '';
    updateCharCount(textEl);

4e — Wire like/reaction to Supabase
Add this handleLike function that wraps the existing toggleLike with a Supabase upsert:
jsasync function handleLike(postId, btn){
  toggleLike(postId, btn); // optimistic UI update immediately

  if(!currentUser) return; // not logged in — UI update only

  const isNowLiked = likedPosts.has(postId);
  try {
    if(isNowLiked){
      await sb.from('reactions').upsert({
        post_id: postId,
        user_id: currentUser.id,
        type: 'heart'
      }, { onConflict: 'post_id,user_id' });
    } else {
      await sb.from('reactions').delete()
        .eq('post_id', postId)
        .eq('user_id', currentUser.id);
    }
  } catch(err){
    console.warn('Reaction sync failed:', err);
  }
}

jsasync function toggleFollow(btn){
  const isFollowing = btn.classList.contains('following');
  btn.classList.toggle('following', !isFollowing);
  btn.textContent = isFollowing ? 'Follow' : '✓ Following';

  if(!currentUser) return;

  const personEl = btn.closest('.follow-person');
  const nameEl = personEl?.querySelector('.fp-name');
  if(!nameEl) return;

  // Look up user by username
  try {
    const { data: profile } = await sb.from('profiles').select('id').eq('username', nameEl.textContent.trim()).single();
    if(!profile) return;

    if(!isFollowing){
      await sb.from('follows').upsert({ follower_id: currentUser.id, following_id: profile.id }, { onConflict: 'follower_id,following_id' });
    } else {
      await sb.from('follows').delete().eq('follower_id', currentUser.id).eq('following_id', profile.id);
    }
  } catch(err){
    console.warn('Follow sync failed:', err);
  }
}
