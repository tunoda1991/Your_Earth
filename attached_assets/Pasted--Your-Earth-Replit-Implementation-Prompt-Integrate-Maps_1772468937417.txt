```typescript
const [currentPage, setCurrentPage] = useState(() => {
  // Check for ?page= query param (from HTML standalone files linking back in)
  const params = new URLSearchParams(window.location.search);
  const pageParam = params.get('page');
  if (pageParam) {
    // Clean the URL so back-nav works cleanly
    window.history.replaceState({}, '', window.location.pathname);
    return pageParam;
  }
  return localStorage.getItem('lastPage') || 'home';
});
```

```javascript
async function yeApi(endpoint, options = {}) {
  const token = localStorage.getItem('auth_token') || window.YE_ANON_KEY;
  const res = await fetch(window.YE_API_BASE + endpoint, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`,
      ...(options.headers || {})
    }
  });
  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    throw new Error(err.error || `API error ${res.status}`);
  }
  return res.json();
}
```

## PART 8 — BACKEND: NEW SUPABASE EDGE FUNCTION ROUTES

The existing `make-server` Edge Function needs 3 new route handlers to support map-related data if needed in the future, and one updated route to support feed-post likes. Add these handlers to the appropriate router file in `supabase/functions/make-server/`:

### 8a. POST `/discussions/:id/like`

```typescript
// Like a discussion post
if (req.method === 'POST' && path.match(/^\/discussions\/[^\/]+\/like$/)) {
  const discussionId = path.split('/')[2];
  const { data: existing } = await supabase
    .from('discussion_likes')
    .select('id')
    .eq('discussion_id', discussionId)
    .eq('user_id', user.id)
    .maybeSingle();

  if (!existing) {
    await supabase.from('discussion_likes').insert({
      discussion_id: discussionId,
      user_id: user.id
    });
    // Increment like count
    await supabase.rpc('increment_likes', { discussion_id: discussionId });
  }
  return new Response(JSON.stringify({ success: true }), { headers: corsHeaders });
}
```

### 8b. DELETE `/discussions/:id/like`

```typescript
if (req.method === 'DELETE' && path.match(/^\/discussions\/[^\/]+\/like$/)) {
  const discussionId = path.split('/')[2];
  await supabase
    .from('discussion_likes')
    .delete()
    .eq('discussion_id', discussionId)
    .eq('user_id', user.id);

  await supabase.rpc('decrement_likes', { discussion_id: discussionId });
  return new Response(JSON.stringify({ success: true }), { headers: corsHeaders });

```

### 8c. Supabase SQL — create `discussion_likes` table and RPC functions

Run this migration in the Supabase SQL editor:

```sql
-- Discussion likes table
CREATE TABLE IF NOT EXISTS discussion_likes (
  id          UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  discussion_id UUID NOT NULL REFERENCES discussions(id) ON DELETE CASCADE,
  user_id     UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(discussion_id, user_id)
);

-- Add likes_count column to discussions if missing
ALTER TABLE discussions ADD COLUMN IF NOT EXISTS likes_count INT DEFAULT 0;
ALTER TABLE discussions ADD COLUMN IF NOT EXISTS reposts_count INT DEFAULT 0;
ALTER TABLE discussions ADD COLUMN IF NOT EXISTS views_count INT DEFAULT 0;
ALTER TABLE discussions ADD COLUMN IF NOT EXISTS comments_count INT DEFAULT 0;

-- Atomic increment/decrement RPCs
CREATE OR REPLACE FUNCTION increment_likes(discussion_id UUID)
RETURNS void LANGUAGE sql AS $$
  UPDATE discussions SET likes_count = likes_count + 1 WHERE id = discussion_id;
$$;

CREATE OR REPLACE FUNCTION decrement_likes(discussion_id UUID)
RETURNS void LANGUAGE sql AS $$
  UPDATE discussions SET likes_count = GREATEST(0, likes_count - 1) WHERE id = discussion_id;
$$;

-- RLS
ALTER TABLE discussion_likes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own likes" ON discussion_likes
  FOR ALL USING (auth.uid() = user_id);
```

## DESIGN RULES — DO NOT VIOLATE

1. **Never change the visual design, layout, color palette, or UX inside any of the three HTML files.** Only the navigation bar HTML is permitted to change in those files.
2. **The three HTML files' internal nav aesthetics must stay native to that file** — `disasters.html` keeps its `Courier New` / `#050a14` nav style; `feed.html` keeps its glassmorphism Inter nav style; `globe_map.html` keeps its existing nav style. Only link targets and item content change.
3. **The React components `InfraMapPage`, `DisasterMapPage`, and `CommunityFeedPage` must be zero-chrome wrappers.** No headers, breadcrumbs, or React UI outside the iframe. The iframe itself provides the full page experience.
4. **Backend calls from feed.html are best-effort / optimistic.** The page must work perfectly even without internet or when unauthenticated. All errors must be swallowed silently or logged as `console.warn`.
5. **The `?page=` query-param routing must not break existing `localStorage` page restoration.** Priority order: `?page=` param → `localStorage.lastPage` → `'home'`.
